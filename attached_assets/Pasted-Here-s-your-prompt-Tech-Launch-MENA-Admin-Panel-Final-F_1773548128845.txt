Backend:

Add GET /admin/export/applications route:

javascriptrouter.get('/admin/export/applications', authenticate, requireAdmin, async (req, res) => {
  const apps = await q(`
    SELECT 
      ea.id, ea.status, ea.stage, ea.notes, ea.created_at,
      u.name as applicant_name, u.email as applicant_email,
      e.name as entity_name
    FROM entity_applications ea
    JOIN users u ON ea.user_id = u.id
    JOIN entities e ON ea.entity_id = e.id
    ORDER BY ea.created_at DESC
  `);
  const csv = [
    'ID,Status,Stage,Entity,Applicant,Email,Notes,Date',
    ...apps.rows.map(r =>
      `${r.id},${r.status},${r.stage},"${r.entity_name}","${r.applicant_name}",${r.applicant_email},"${r.notes || ''}",${r.created_at}`
    )
  ].join('\n');
  res.setHeader('Content-Type', 'text/csv');
  res.setHeader('Content-Disposition', 'attachment; filename="applications.csv"');
  res.send(csv);
});
