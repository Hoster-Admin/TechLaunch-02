FIX 2 — Lock CORS to Production Domain
File: backend/src/server.js
Replace the current CORS config with:
javascriptconst allowedOrigins = process.env.NODE_ENV === 'production'
  ? [
      'https://tlmena.com',
      'https://www.tlmena.com'
    ]
  : [
      'http://localhost:3000',
      'http://localhost:5173',
      /\.replit\.dev$/,
      /\.replit\.app$/
    ];

app.use(cors({
  origin: function(origin, callback) {
    if (!origin) return callback(null, true);
    const allowed = allowedOrigins.some(o =>
      typeof o === 'string' ? o === origin : o.test(origin)
    );
    if (allowed) {
      callback(null, true);
    } else {
      callback(new Error(`CORS blocked: ${origin}`));
    }
  },
  credentials: true
}));
This allows Replit URLs in development but locks to tlmena.com only in production.
