STEP 6 — Validation Rules
Add these validation checks in the import route before inserting:
javascriptconst validateRow = (row, index) => {
  const errors = [];

  if (!row.name?.trim()) {
    errors.push(`Row ${index}: name is required`);
  }

  const validTypes = ['startup', 'accelerator', 'investor', 'venture_studio'];
  if (!validTypes.includes(row.type?.trim())) {
    errors.push(`Row ${index}: type must be one of: ${validTypes.join(', ')}`);
  }

  if (!row.country?.trim()) {
    errors.push(`Row ${index}: country is required`);
  }

  if (row.website && !row.website.startsWith('http')) {
    errors.push(`Row ${index}: website must start with http`);
  }

  if (row.founded_year && isNaN(parseInt(row.founded_year))) {
    errors.push(`Row ${index}: founded_year must be a number`);
  }

  if (row.portfolio_count && isNaN(parseInt(row.portfolio_count))) {
    errors.push(`Row ${index}: portfolio_count must be a number`);
  }

  return errors;
};
